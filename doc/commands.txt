COMMAND :=
	VARIABLE_ASSIGNMENT ';'
|	IF_COMMAND
|	'{' HELPER_COMMAND_LIST '}'
|	'{'	'}'
;

HELPER_COMMAND_LIST :=
	COMMAND
|	COMMAND HELPER_TYPE_LIST
;

VARIABLE_ASSIGNMENT :=
	LVALUE_EXPRESSION '=' EXPRESSION
;

IF_COMMAND :=
	if '(' EXPRESSION ')' COMMAND
|	if '(' EXPRESSION ')' COMMAND else COMMAND
;

FOR_COMMAND :=
	for '(' EXPRESSION ')' COMMAND
	
BREAK_COMMAND := 
	break INT_LITERAL ';'
|	break ';'
;
	
CONTINUE_COMMAND :=
	continue ';'
;

RETURN_COMMAND :=
	return EXPRESSION ';'
|	return ';'
;
