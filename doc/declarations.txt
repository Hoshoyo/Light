TOP_LEVEL :=
	HELPER_DECLARATION_LIST
;

HELPER_DECLARATION_LIST :=
	DECLARATION
|	DECLARATION HELPER_DECLARATION_LIST
;

DECLARATION := 
	DECLARATION_PROCEDURE
|	DECLARATION_VARIABLE
|	DECLARATION_STRUCT
|	DECLARATION_CONSTANT
|	DECLARATION_UNION
|	DECLARATION_ENUM
;

HELPER_DECLARATION_VARIABLE_LIST :=
	DECLARATION_VARIABLE
|	DECLARATION_VARIABLE ',' HELPER_DECLARATION_VARIABLE_LIST
;

HELPER_DECLARATION_ARGUMENTS_LIST :=
	DECLARATION_VARIABLE
|	DECLARATION_VARIABLE ';' HELPER_DECLARATION_ARGUMENTS_LIST 

HELPER_DECLARATION_CONSTANT_LIST :=
	DECLARATION_CONSTANT
|	IDENTIFIER
|	DECLARATION_CONSTANT ',' HELPER_DECLARATION_CONSTANT_LIST
|	IDENTIFIER ',' HELPER_DECLARATION_CONSTANT_LIST
;

DECLARATION_PROCEDURE :=
	IDENTIFIER ':' ':' '('  ')' '->' TYPE '{' HELPER_COMMAND_LIST '}'
|	IDENTIFIER ':' ':' '(' HELPER_DECLARATION_VARIABLE_LIST ')' '->' TYPE '{' HELPER_COMMAND_LIST '}'
;

DECLARATION_VARIABLE :=
	IDENTIFIER ':' TYPE
|	IDENTIFIER ':' TYPE '=' LITERAL
;

DECLARATION_STRUCT :=
	IDENTIFIER ':' ':' struct '{' HELPER_DECLARATION_ARGUMENTS_LIST '}'
;

DECLARATION_UNION :=
	IDENTIFIER ':' ':' union '{' HELPER_DECLARATION_ARGUMENTS_LIST '}'
;

DECLARATION_ENUM :=
	IDENTIFIER ':' ':' enum '{' HELPER_DECLARATION_CONSTANT_LIST '}'
|	IDENTIFIER ':' TYPE ':' enum '{' HELPER_DECLARATION_CONSTANT_LIST '}'
;

DECLARATION_CONSTANT :=
	IDENTIFIER ':' ':' LITERAL
|	IDENTIFIER ':' ':' CONSTANT
|	IDENTIFIER ':' TYPE ':' LITERAL
|	IDENTIFIER ':' TYPE ':' CONSTANT
;
